CREATE TABLE `clinic`.`medical_appiontement` (
  `APPIONTEMENT_ID` INT NOT NULL AUTO_INCREMENT,
  `PATIENT_ID` INT NULL,
  `DOCTOR_ID` INT NULL,
  `APPIONTEMENT_DATE` DATETIME NULL,
  `ISACTIVE` VARCHAR(3) NULL,
  `REASON` VARCHAR(200) NULL,
  PRIMARY KEY (`APPIONTEMENT_ID`),
  UNIQUE INDEX `APPIONTEMENT_ID_UNIQUE` (`APPIONTEMENT_ID` ASC) VISIBLE,
  INDEX `MEDICAL_APPIONTEMENT_FK1_idx` (`PATIENT_ID` ASC) VISIBLE,
  INDEX `MEDICAL_APPIONTEMENT_FK2_idx` (`DOCTOR_ID` ASC) VISIBLE,
  CONSTRAINT `MEDICAL_APPIONTEMENT_FK1`
    FOREIGN KEY (`PATIENT_ID`)
    REFERENCES `clinic`.`medical_patient` (`PATIENT_ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `MEDICAL_APPIONTEMENT_FK2`
    FOREIGN KEY (`DOCTOR_ID`)
    REFERENCES `clinic`.`medical_doctor` (`DOCTOR_ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
